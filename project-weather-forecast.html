Project Overview

The WeatherEase application is a Python-based desktop app that fetches and displays real-time weather data from the OpenWeatherMap API. It provides users with current weather conditions, temperature, humidity, wind speed, and a 5-day forecast in an intuitive Tkinter GUI.

Code Snippet - API Integration

Here's the core Python code that fetches weather data from the API:


import tkinter as tk
from tkinter import messagebox
import threading
import requests
from datetime import datetime, timedelta
from timezonefinder import TimezoneFinder
import pytz
from PIL import Image,ImageTk

# Function to get coordinates (latitude and longitude) from city name
def get_coordinates(city, api_key):
    geocoding_url = f"http://api.openweathermap.org/geo/1.0/direct?q={city}&limit=1&appid={api_key}"
    response = requests.get(geocoding_url)
    
    if response.status_code == 200:
        data = response.json()
        if data:
            lat = data[0]['lat']
            lon = data[0]['lon']
            return lat, lon
        else:
            return None
    else:
        return None

# Function to get weather data using coordinates
def get_weather():
    city = search.get()
    if city:
        api_key = "cb57a3ab8c2e256e264e302a3bf8aa03"
        coordinates = get_coordinates(city, api_key)
        
        if coordinates:
            lat, lon = coordinates
            # Get current weather data
            weather_url = f"https://api.openweathermap.org/data/2.5/weather?lat={lat}&lon={lon}&appid={api_key}"
            weather_response = requests.get(weather_url)
            
            # Get 5-day forecast data
            forecast_url = f"https://api.openweathermap.org/data/2.5/forecast?lat={lat}&lon={lon}&appid={api_key}"
            forecast_response = requests.get(forecast_url)
            
            if weather_response.status_code == 200 and forecast_response.status_code == 200:
                weather_data = weather_response.json()
                forecast_data = forecast_response.json()
                
                main = weather_data['main']
                weather = weather_data['weather'][0]
                temperature = main['temp'] - 273.15
                kelvin = temperature * 9/5 + 32
                humidity = main['humidity']
                wind = weather_data['wind']['speed']
                description = weather['description']
                
                lon = weather_data['coord']['lon']
                lat = weather_data['coord']['lat']
                tf = TimezoneFinder()
                result = tf.timezone_at(lng=lon, lat=lat)
                home = pytz.timezone(result)
                local_time = datetime.now(home).strftime("%Y/%m/%d %I:%M:%p")
                
                location['text'] = weather_data['name']
                timelbl['text'] = local_time
                temperature_lbl['text'] = f"{temperature:.2f}°C"
                kelvin_lbl['text'] = f"{kelvin:.2f}°F"
                humidity_lbl['text'] = f"{humidity}%"
                pressure_lbl['text'] = f"{main['pressure']} hPa"
                description_lbl['text'] = f"{description.capitalize()}"
                wind_lbl['text'] = f"{wind} m/s"
                sunrise_time = datetime.fromtimestamp(weather_data['sys']['sunrise']).strftime('%Y/%m/%d %I:%M:%p')
                sunset_time = datetime.fromtimestamp(weather_data['sys']['sunset']).strftime('%Y/%m/%d %I:%M:%p')
                sunrise_lbl['text'] = f"Sunrise: \n{sunrise_time}"
                sunset_lbl['text'] = f"Sunset: \n{sunset_time}"
                
                forecast_info = "5-Day Forecast\n\n"
                daily_data = {}
                for entry in forecast_data['list']:
                    date = entry['dt_txt'].split(" ")[0]
                    if date not in daily_data:
                        daily_data[date] = {'temp': [], 'description': []}
                    daily_data[date]['temp'].append(entry['main']['temp'] - 273.15)
                    daily_data[date]['description'].append(entry['weather'][0]['description'])
                
                forecast_info = ""
                for i, (date, data) in enumerate(daily_data.items()):
                    if i >= 5:
                        break
                    avg_temp = sum(data['temp']) / len(data['temp'])
                    description = max(set(data['description']), key=data['description'].count).capitalize()
                    day_name = (datetime.now() + timedelta(days=i)).strftime("%A")
                    forecast_info += f"{day_name}: {avg_temp:.2f}°C, {description}\n"
                
                forecast_label.config(text=forecast_info)
            else:
                messagebox.showerror("Error", "Weather data not found.")
        else:
            messagebox.showerror("Error", "City not found.")
    else:
        messagebox.showwarning("Input Error", "Please enter a city name.")
